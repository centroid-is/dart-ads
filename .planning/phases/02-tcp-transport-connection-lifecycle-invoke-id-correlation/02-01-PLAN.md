abstract interface class AdsTransport {
  Future<void> connect(String host, int port);   // throws SocketException on refuse
  void add(List<int> bytes);                      // outbound; non-blocking, buffered
  Stream<Uint8List> get inbound;                  // single-subscription byte chunks
  Future<void> close();                           // flush() then destroy()
}

<!-- dart:io Socket IS a Stream<Uint8List> and IS an IOSink: -->
<!--   Socket.connect(host, port) -> Future<Socket> -->
<!--   socket.setOption(SocketOption.tcpNoDelay, true) -->
<!--   socket.add(bytes); await socket.flush(); socket.destroy(); -->
<!--   'close()' alone only half-closes (sends FIN) ‚Äî use flush()+destroy(). -->
</interfaces>
